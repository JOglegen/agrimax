2
Extract Boundary Data from Response

Once you find where the geometry is, extract it:

async function jdFetchFields() { try { const response = await fetch( 'https://api.deere.com/platform/organizations/' + jdOrgId + '/fields', { headers: { 'Authorization': 'Bearer ' + jdToken } } ); const jdResponse = await response.json(); const jdFields = jdResponse.values || jdResponse.fields || []; let imported = 0; for (const jdField of jdFields) { // TRY different possible locations for geometry let geometry = null; if(jdField.geometry) { geometry = jdField.geometry; } else if(jdField['@geometry']) { geometry = jdField['@geometry']; } else if(jdField.boundary) { geometry = jdField.boundary; } else if(jdField.shape) { geometry = jdField.shape; } console.log(`${jdField.name}: ${geometry ? 'Has geometry' : 'NO geometry'}`); const fieldData = { id: jdField.id, name: jdField.name, farm_id: CURRENT_FARM_ID, acres: jdField.acres || jdField.area, geojson: geometry, // Store geometry source: 'john_deere' }; const { error } = await supa.from('fields').upsert(fieldData); if(!error) imported++; } toast(`✅ Imported ${imported} fields with boundaries`); await loadAll(); } catch(e) { err("Import failed: " + e.message); } }
3
Verify Database Schema

Make sure your fields table can store geometry:

-- Check schema PRAGMA table_info(fields); -- Add column if missing ALTER TABLE fields ADD COLUMN geojson JSON; -- Verify stored data SELECT id, name, CASE WHEN geojson IS NOT NULL THEN '✓' ELSE '✗' END as has_geometry FROM fields LIMIT 5;

Render boundaries on map:

function displayFieldsOnMap() { const fields = DATA.fields || []; fields.forEach(field => { if(!field.geojson) return; L.geoJSON(field.geojson, { style: { color: '#1e5631', weight: 2, fillOpacity: 0.2 }, onEachFeature: (feature, layer) => { layer.bindPopup(field.name); layer.on('click', () => { CURRENT_FIELD_ID = field.id; renderFieldDetails(field.id); }); } }).addTo(map); }); }

  • Database has geojson column
